PropertyGrid

Status: Stable
Category: collections
Import: drawui
Emitted class(es): PropertyGrid, PropertyGrid-row

Key/value property editor grid.

Import

import { PropertyGrid } from "drawui";

Examples

PropertyGrid

const nameField = new InputText().setValue("Beam-12");
  const typeSelect = new InputDropdown().setOptions({
    IfcBeam: "IfcBeam",
    IfcColumn: "IfcColumn",
    IfcSlab: "IfcSlab",
  }).setValue("IfcBeam");
  const notes = new InputTextArea().setValue("Primary span member");
  notes.dom.rows = 2;
  const visible = new Checkbox(true).addClass("Card-checkbox");
  const length = new InputNumber(12.5).setRange(0, 100).setUnit("m");
  const count = new InputInteger(4).setRange(0, 99);
  const opacity = new Slider(0.75).setRange(0, 1).setStep(0.01).setPrecision(2);
  const color = new InputColor().setValue("#70ba35");
  const created = new InputDate(new Date());
  const progress = new ProgressBar(0.45);

  return new PropertyGrid({
    compact: true,
    title: "All value types",
    rows: {
      staticText: { label: "Text", value: "IfcBeam" },
      name: { label: "Input", value: nameField },
      type: { label: "Select", value: typeSelect },
      notes: { label: "Textarea", value: notes },
      visible: { label: "Checkbox", value: visible },
      length: { label: "Number", value: length },
      count: { label: "Integer", value: count },
      opacity: { label: "Slider", value: opacity },
      color: { label: "Color", value: color },
      created: { label: "Date", value: created },
      progress: { label: "Progress", value: progress },
    },
  });
