Form

Status: Stable
Category: inputs
Import: drawui
Emitted class(es): None by default

Form wrapper with submit helpers.

Import

import { Form } from "drawui";

Examples

Form

const nameInput = new InputText();
  nameInput.dom.placeholder = "Jane Doe";
  const emailInput = new InputText();
  emailInput.dom.placeholder = "jane@example.com";
  emailInput.dom.type = "email";
  const notes = new InputTextArea();
  notes.dom.placeholder = "Optional notes…";
  const roleSelect = new InputDropdown().setOptions({ user: "User", admin: "Admin", editor: "Editor" });
  const actions = new StackPanel({ isVertical: false })
    .gap("0.5rem")
    .add(new Button("Cancel").addClass("secondary"))
    .add(new Button("Save").addClass("primary"));
  return new Form()
    .setStyle("display", ["flex"])
    .setStyle("flexDirection", ["column"])
    .gap("0.5rem")
    .add(new Label("Name"))
    .add(nameInput)
    .add(new Label("Email"))
    .add(emailInput)
    .add(new Label("Role"))
    .add(roleSelect)
    .add(new Label("Notes"))
    .add(notes)
    .add(
      new StackPanel({ isVertical: false })
        .gap("0.5rem")
        .add(new Checkbox(true).addClass("Card-checkbox"))
        .add(new TextBlock("Email me updates")),
    )
    .add(new Spacer("0.5rem"))
    .add(actions);
