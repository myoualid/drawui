WorkspacePanel

Status: Stable
Category: panels
Import: drawui
Emitted class(es): WorkspacePanel, CollapsiblePanel, ContentPanel, Header, PanelContent, RibbonBar

{@link CollapsiblePanel} for {@link WorkspaceLayout} regions.

Owns docked chrome (collapse + optional undock). The layout system only mounts, moves, and tracks panels — it does not create or style this component.

Source: src/components/panels/WorkspacePanel.js

Import

import { WorkspacePanel } from "drawui";

Examples

Usage

const panel = new WorkspacePanel({
  context, position: 'right', tabId: 'settings', tabLabel: 'Settings',
});
panel.add(new TextBlock('Settings'));
panel.show();

WorkspacePanel

const wrap = new StackPanel({ isVertical: true }).gap("0.75rem");

  let panel = null;

  const openButton = new Button("Open floating tab panel")
    .addClass("primary")
    .onClick(() => {
      if (!panel) {
        panel = new WorkspacePanel({
          context: { ui: { model: {} } },
          tabId: "gallery-demo-tab",
          tabLabel: "Demo panel",
          icon: "tab",
          title: "Floating tab panel",
          startFloating: true,
          floatingStyles: { width: "320px", height: "260px" },
        });
        panel.add(
          new StackPanel({ isVertical: true })
            .gap("0.5rem")
            .setStyle("padding", ["0.75rem"])
            .add(new TextBlock("WorkspacePanel content — dock buttons appear when a WorkspaceLayout context is provided."))
            .add(new Badge("startFloating: true")),
        );
      }
      panel.show();
    });

  wrap.add(openButton);
  wrap.add(
    new Caption(
      "Without a WorkspaceLayout the panel opens as a floating window; inside a workspace it docks as a CollapsiblePanel (undock transfers content into a FloatingWindow).",
    ),
  );
  return wrap;
