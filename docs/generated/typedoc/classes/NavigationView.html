    Class NavigationView

    Example: NavigationView

    const host = new Container().setStyle("height", ["260px"]).setStyle("overflow", ["hidden"]);
      const layout = new NavigationView({
        sidebarWidth: "180px",
        sidebarMinWidth: "140px",
        sidebarMaxWidth: "320px",
        sidebarResizable : true,
      });
      layout.setSidebarTitle("Explorer");
      layout.setSidebarContent(
        new StackPanel({ isVertical: true })
          .gap("0.25rem")
          .add(new TextBlock("Site"))
          .add(new TextBlock("Building")),
      );
      layout.setMainContent(
        new StackPanel({ isVertical: true })
          .gap("0.5rem")
          .add(new Heading(3, "Viewport"))
          .add(new TextBlock("Main content")),
      );
      host.add(layout);
      return host;
