    Class FloatingWindow

    Free-floating window with docking controls.

    Example: FloatingWindow

    return new Button("Open FloatingWindow").addClass("secondary").onClick(() => {
        document.querySelectorAll(".FloatingWindow").forEach((node) => {
          if (node.parentElement === document.body) node.remove();
        });
        const panel = new FloatingWindow({ title: "Overlay", width: "320px" });
        panel.setContent(
          new StackPanel({ isVertical: true })
            .gap("0.5rem")
            .add(new Heading(3, "FloatingWindow"))
            .add(new TextBlock("Free-floating overlay panel")),
        );
        panel.dom.style.height = "auto";
        panel.dom.style.maxHeight = "70vh";
        panel.show();
      });
